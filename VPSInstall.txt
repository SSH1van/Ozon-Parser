#!/bin/bash

cd /home
mkdir Ozon-Parser

yes | sudo add-apt-repository ppa:openjdk-r/ppa
sudo apt update && apt upgrade -y
sudo apt install openjdk-23-jdk -y
sudo apt install htop -y

wget https://dl.google.com/linux/direct/google-chrome-stable_current_amd64.deb
sudo dpkg -i google-chrome-stable_current_amd64.deb
sudo apt --fix-broken install -y
sudo apt update && apt upgrade -y

wget https://storage.googleapis.com/chrome-for-testing-public/132.0.6834.159/linux64/chromedriver-linux64.zip
sudo apt install unzip
unzip chromedriver-linux64.zip
chmod +x chromedriver-linux64/chromedriver


# chromedrivers - https://googlechromelabs.github.io/chrome-for-testing/#stable
